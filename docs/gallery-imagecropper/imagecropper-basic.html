Example: Augmenting an image with an ImageCropper

This example demonstrates how to instantiate a simple ImageCropper. Try clicking on the button to get the coordinates of the crop area.

Complete Example Source

<div id="demo" class="yui3-skin-sam">
  <button id="save" class="yui3-button" type="button">Save crop area</button>
  <img id="crop" src="http://farm8.staticflickr.com/7172/6748677447_a444f2cbce.jpg" alt="Bass player">
</div>

<script type="text/javascript">
YUI().use('gallery-imagecropper', function(Y) {
    
    // Create a new instance of ImageCropper from markup          
    var imagecropper = new Y.ImageCropper({
      srcNode: "#crop",
      width: 335,
      height: 500
    }).render();

    Y.one('#save').on('click', function () {
      var coords = imagecropper.getCropCoords()
      var msg = 'left: '   + coords.left  + '\n';
         msg += 'top: '    + coords.top   + '\n';
         msg += 'width: '  + coords.width + '\n';
         msg += 'height: ' + coords.height;
      alert(msg);
    });
    
});
</script>
